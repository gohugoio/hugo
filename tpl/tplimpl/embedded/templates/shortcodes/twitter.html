{{- $pc := .Page.Site.Config.Privacy.Twitter -}} {{- if not $pc.Disable -}} {{- if $pc.Simple -}} {{- template "_internal/shortcodes/twitter_simple.html" . -}} {{- else -}} {{- $id := or (.Get "id") "" -}} {{- $user := or (.Get "user") "" -}} {{- if and $id $user -}} {{- template "render-tweet" (dict "id" $id "user" $user "dnt" $pc.EnableDNT "name" .Name "position" .Position) -}} {{- else -}} {{- errorf "The %q shortcode requires two named parameters: user and id. See %s" .Name .Position -}} {{- end -}} {{- end -}} {{- end -}} {{- define "render-tweet" -}} {{- $url := printf "https://twitter.com/%v/status/%v" .user .id -}} {{- $query := querify "url" $url "dnt" .dnt -}} {{- $request := printf "https://publish.twitter.com/oembed?%s" $query -}} {{- with resources.GetRemote $request -}} {{- with .Err -}} {{- errorf "%s" . -}} {{- else -}} {{- (. | transform.Unmarshal).html | safeHTML -}} {{- end -}} {{- else -}} {{- warnidf "shortcode-twitter-getremote" "The %q shortcode was unable to retrieve the remote data. See %s" .name .position -}} {{- end -}} {{- end -}}
